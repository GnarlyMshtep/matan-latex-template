\begin{dmath*}
\inabs{
    \Pr \insqr{ #1 \; \left| \;
    \begin{aligned}

    \end{aligned}
    }\right.

    \Pr \insqr{ #3 \; \left| \;
    \begin{aligned}
        #4
    \end{aligned}
    }\right.
    } = #5

}
